#### 选择排序
```javascript
var arr=[88,55,88,55,3,6]

function selectionSort(arr) {
    // 选择排序
    let minIndex,temp

    for (let i = 0; i < arr.length; i++) {
        minIndex = i
        for (let j = i+1; j < arr.length; j++) {
            if (arr[j]<arr[minIndex]) { //寻找最小数
                minIndex=j //保存最小的索引
            }
        }
        temp = arr[i]  //将当前值与最小值交换数据
        arr[i] = arr[minIndex]
        arr[minIndex] = temp
    }
    return arr
}

console.log(selectionSort(arr));
```
